    \section*{Frequently used terms}%
    \vskip -2em
    \begin{tabularx}{\textwidth}{lX}
    	Edit distance		& Minimal cost to edit one sequence to another \\
    	Alignment 			& Correspondence between letters of two sequences \\[-3pt]
		\,$\sbullet$ Global 			& Fully aligning two sequences to each other \\[-3pt]
		\,$\sbullet$ Semi-global		& Fully aligning a query somewhere in a reference \\[-3pt]
		\,$\sbullet$ Reads alignment 		& Multiple semi-global alignments to the same reference \\
    	Dynamic programming & Optimization technique by reusing partial results \\[-3pt]
    	\,$\sbullet$ Explored state& Pushed to the priority queue \\[-3pt]
    	\,$\sbullet$ Expanded state& Popped from the priority queue \\
		\dijkstra's algorithm & Uninformed shortest path algorithm \\
    	\A algorithm 		& Generalization of \dijkstra, informed by a heuristic \\[-3pt]
		\,$\sbullet$ Admissible heuristic		& Lower bound on the remaining shortest distance \\
    	Graph reference		& Compressed representation of multiple references \\
    	Trie index 			& Tree structure with short reference subsequences \\
    	Seed heuristic		& Our proposed heuristic function for \A \\[-3pt]
		\,$\sbullet$ Chaining & Requiring that all matches are consecutive  \\[-3pt]
		\,$\sbullet$ Inexact matching & Allowing a number of errors when matching a seed \\[-3pt]
		\,$\sbullet$ Potential & Maximal value of the heuristic \\
    	Performance 		& Runtime and memory footprint \\
    	Scaling	 			& Change (not necessarily asymptotic) with bigger input \\
    \end{tabularx}

    \section*{Frequently used symbols}%
    \vskip -2em
    \begin{tabularx}{\textwidth}{lX}
		$\Sigma = \{A,C,G,T\}$ 			& alphabet \\
		$A, B \in \Sigma^*$				& sequences \\
		$q \in \Sigma^m$				& query/read \\
%    	edit operation		& insertion, deletion or substitution of a single letter \\
		$\cmatch, \csubst, \cins, \cdel \geq 0$ & edit costs for match, insertion, deletion and substitution of a single letter \\[5pt]
		$ed(A,B)$						& edit distance \\
		$\RG=(\RGV,\RGE)$               & reference graph \\[5pt]
%	Position in read & $q[i] \in \Sigma$, $i \in \{0,\dots,m-1\}$\\	
		$\st{u}{i}$ 					& state (subalignment of $A$)\\
		$\AG=(\AGV,\AGE)$ 				& alignment graph on states $\AGV$\\
		$D$								& trie depth \\
		$g$								& best distance from start \\
		$h\colon V \to \Costs$ 			& heuristic function \\
		$f = g + h$						& estimated total distance \\
		$h\st{u}{i}$					& seed heuristic \\
		$s\in \seeds$					& seeds \\
		$k$ 							& seed length \\
		$r$								& seed potential \\
		$P\st ij = r \cdot \abs{\seeds_{\geq i}}$ & state potential \\
		$m \in \matches_s$				& matches of seed $s$ \\
		$\matchcost(m) \in [0, r)$ 		& cost of match $m$ \\
		$\gamma(m, m')$ 				& chaining cost \\
		$\gapcost(\st ij, \st{i'}{j'}):=\abs{(i'{-}i) {-} (j'{-}j)}$ & gap cost \\
		$\seedcost(u,v) = r\cdot \substr \seeds uv$ & seed cost \\
		$\gscost = \max(\gapcost, \seedcost)$ & gap-seed cost \\
		$\bluecrumb{}, \yellowcrumb{}, \violetcrumb{}, \greencrumb{}$ & crumbs on nodes for match $m$ \\
    \end{tabularx}
